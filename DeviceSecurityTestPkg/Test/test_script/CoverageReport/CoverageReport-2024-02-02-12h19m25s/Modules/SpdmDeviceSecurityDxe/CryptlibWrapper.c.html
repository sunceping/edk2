void *
libspdm_sha256_new (
  void
  )
{
  size_t  CtxSize;
  void    *HashCtx;

  HashCtx = NULL;
  CtxSize = Sha256GetContextSize ();
  HashCtx = AllocatePool (CtxSize);

  return HashCtx;
}

void
libspdm_sha256_free (
  void  *Sha256Ctx
  )
{
  if (Sha256Ctx != NULL) {
    FreePool (Sha256Ctx);
    Sha256Ctx = NULL;
  }
}

bool
libspdm_rsa_pkcs1_sign_with_nid (
  void           *rsa_context,
  size_t         hash_nid,
  const uint8_t  *message_hash,
  size_t         hash_size,
  uint8_t        *signature,
  size_t         *sig_size
  )
{
  switch (hash_nid) {
    case CRYPTO_NID_SHA256:
      if (hash_size != SHA256_DIGEST_SIZE) {
        return FALSE;
      }

      break;

    case CRYPTO_NID_SHA384:
      if (hash_size != SHA384_DIGEST_SIZE) {
        return FALSE;
      }

      break;

    case CRYPTO_NID_SHA512:
      if (hash_size != SHA512_DIGEST_SIZE) {
        return FALSE;
      }

      break;

    default:
      return FALSE;
  }

  return RsaPkcs1Sign (
           rsa_context,
           message_hash,
           hash_size,
           signature,
           sig_size
           );
}
